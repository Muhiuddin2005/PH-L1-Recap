#<!DOCTYPE html>
<html>
<head>
  <script src="https://cdn.tailwindcss.com"></script>

  <style>
    /* Tag selector (normal CSS, works with Tailwind) */
    p {
      color: blue;                        /* all <p> tags become blue */
    }

    /* ID selector */
    #main {
      color: red;                         /* used for ONE unique element */
    }

    /* Class selector (Tailwind classes) */
    .title {
      @apply text-2xl font-bold;          /* Tailwind applied through CSS */
    }

    /* Universal selector */
    * {
      margin: 0;                          /* affects everything */
    }

    /* Grouping selector */
    h1, h2, p {
      font-family: Arial;                 /* applies to all three */
    }

    /* Attribute selector */
    input[type="text"] {
      border: 2px solid green;            /* targets only text inputs */
    }
  </style>
</head>

<body class="p-4">

  <h1 class="title text-blue-700">Heading using class</h1> <!-- class for reuse -->

  <h2 id="main" class="text-xl">Heading using ID</h2>      <!-- ID for unique element -->

  <p class="mt-2">Paragraph using tag selector + Tailwind</p>

  <input type="text" class="mt-3 p-2" placeholder="Text input"> <!-- attribute selector -->

  <input type="password" class="mt-3 p-2" placeholder="Password"> <!-- not selected -->

</body>

#<!DOCTYPE html>
<html>
<head>
  <script src="https://cdn.tailwindcss.com"></script>

  <style>
    /* Combined selector (only <p> with class="info") */
    p.info {
      @apply bg-yellow-200;            /* Tailwind apply */
    }

    /* Attribute selector */
    input[type="email"] {
      @apply border-2 border-blue-500; /* style only email input */
    }

    /* div selector */
    div {
      @apply p-4 bg-gray-200;          /* block container */
    }

    /* span selector */
    span {
      @apply text-green-600 italic;    /* inline highlight */
    }
  </style>
</head>

<body class="p-5 space-y-4">

  <!-- Multiple classes (just add more Tailwind utilities) -->
  <p class="text-2xl text-red-500 font-bold">
    Big and red text (multiple Tailwind classes)
  </p>

  <!-- Combined selector applies from CSS above -->
  <p class="info p-2">
    Paragraph with combined selector (p.info)
  </p>

  <!-- Attribute selector (input[type="email"]) -->
  <input type="email" class="p-2" placeholder="Enter email">

  <!-- Div: block element -->
  <div>
    This is inside a div (styled with Tailwind in CSS).
  </div>

  <!-- Span: inline element -->
  <p>
    This is a <span>styled part</span> inside a sentence.
  </p>

</body>
</html>
#In HTML, an attribute is extra information you can add to an HTML element to modify its behavior or provide additional details.
<!-- 'type' and 'placeholder' are attributes -->
<input type="text" placeholder="Enter your name">

<!-- 'href' is an attribute -->
<a href="https://example.com">Visit Example</a>

<!-- 'src' and 'alt' are attributes -->
<img src="photo.jpg" alt="A photo">

#/* DISPLAY NONE */
.hide {
  display: none;       /* element is removed from page layout completely */
}

/* VISIBILITY HIDDEN */
.hidden {
  visibility: hidden;  /* element is invisible but still takes up space */
}

/* INLINE */
.inline {
  display: inline;     /* element flows like text, width/height ignored.An inline element does 
  not start on a new line; it stays on the same line as the previous content.*/
}

/* BLOCK */
.block {
  display: block;      /* element takes full width, starts on new line */
}

/* INLINE-BLOCK */
.inline-block {
  display: inline-block; /* flows inline but can set width/height */
}

/* BOX SHADOW */
.shadow-box {
  width: 150px;
  height: 100px;
  background: lightblue;
  box-shadow: 5px 5px 10px gray;  /* horizontal, vertical, blur, color */
}
