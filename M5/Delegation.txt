  // TEACHER LISTENING TO THE WHOLE CLASS (Event Delegation)
  container.addEventListener("click", function(event){

      // event.target = who actually raised hand (clicked)
      if(event.target.classList.contains("child-btn")){
          alert("You clicked: " + event.target.innerText);

  });

  // THIS SHOWS EVENT BUBBLING
  container.addEventListener("click", function(){
      console.log("The container also heard the click");
  });
</script>
