## **ProductManagement.jsx**

```jsx
import React, { useState } from 'react';
import ProductForm from './productForm';
import ProductTable from './ProductTable';

const ProductManagement = () => {

    // Parent state: stores all products as an array
    const [products, setProducts] = useState([]);

    // This function receives a new product from ProductForm
    const handleAddProduct = (newProduct) => {
        // Create a new array adding this product
        const newProducts = [...products, newProduct];

        // Update the products state
        setProducts(newProducts);
    };

    return (
        <div>
            {/* 
                Passing handleAddProduct to child ProductForm
                so the child can send data back to parent 
            */}
            <ProductForm handleAddProduct={handleAddProduct} />

            {/* Passing products list to ProductTable */}
            <ProductTable products={products} />
        </div>
    );
};

export default ProductManagement;
```

## **ProductForm.jsx**

```jsx
import React, { useState } from 'react';

const ProductForm = ({ handleAddProduct }) => {

    // For showing error messages
    const [error, setError] = useState('');

    // Handles form submit
    const handleProductSubmit = (e) => {
        e.preventDefault(); // prevent page reload

        // Getting values from input fields
        const name = e.target.name.value;
        const price = e.target.price.value;
        const quantity = e.target.quantity.value;

        // Simple validations
        if (name.length === 0) {
            setError('Please provide a product name');
            return;   // stop function
        }
        else if (price.length === 0) {
            setError('Please provide a price');
            return;
        }
        else if (price < 0) {
            setError('Price can not be negative');
            return;
        }
        else {
            setError(''); // remove error
        }

        // Create product object
        const newProduct = {
            name,
            price,
            quantity
        };

        // Send this product to parent component
        handleAddProduct(newProduct);

        // OPTIONAL: clear the form if you want
        // e.target.reset();
    };

    return (
        <div>
            <h3>Add a Product</h3>

            {/* Form submission */}
            <form onSubmit={handleProductSubmit}>

                <input type="text" name="name" placeholder="Product Name" />
                <br />

                <input type="text" name="price" placeholder="Product Price" />
                <br />

                <input type="text" name="quantity" placeholder="Product Quantity" />
                <br />

                <input type="submit" value="Submit" />
            </form>

            {/* Error message */}
            <p style={{ color: 'red' }}>
                <small>{error}</small>
            </p>
        </div>
    );
};

export default ProductForm;
```
