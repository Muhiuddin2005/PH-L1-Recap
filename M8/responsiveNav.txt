# âœ… **Final Enhanced Navbar Code (With Animations + Relative/Absolute + Hover/Click Options)**

```jsx
import React, { useState } from "react";
import { Menu, X } from "lucide-react"; 

export default function Navbar() {
  const [isOpen, setIsOpen] = useState(false);

  // -------------------------
  // ðŸŽ› TOGGLE THIS TO ENABLE HOVER
  const enableHoverOpen = false; 
  // -------------------------

  return (
    <nav className="bg-white shadow-md fixed top-0 left-0 w-full z-50">
      {/* Parent container for absolute menu */}
      <div className="relative max-w-6xl mx-auto">

        <div className="flex items-center justify-between px-4 py-3 md:px-8">
          {/* Logo */}
          <h1 className="text-xl font-bold">MyLogo</h1>

          {/* Desktop Menu */}

            <li className="cursor-pointer hover:text-blue-600">Home</li>
            <li className="cursor-pointer hover:text-blue-600">About</li>
            <li className="cursor-pointer hover:text-blue-600">Services</li>
            <li className="cursor-pointer hover:text-blue-600">Contact</li>
          </ul>

          {/* Mobile Hamburger Icon */}
          <button
            className="md:hidden"
            onClick={() => {
              if (!enableHoverOpen) setIsOpen(!isOpen);
            }}
            onMouseEnter={() => {
              if (enableHoverOpen) setIsOpen(true);
            }}
            onMouseLeave={() => {
              if (enableHoverOpen) setIsOpen(false);
            }}
          >
            {isOpen ? <X size={28} /> : <Menu size={28} />}
          </button>
        </div>

        {/* ------------------------------------
            ðŸ“± Mobile Dropdown Menu

        ------------------------------------- */}
        <ul
          className={`

            transition-all duration-300 ease-in-out 
            overflow-hidden 
            ${isOpen ? "max-h-60 opacity-100" : "max-h-0 opacity-0"}
          `}
          onMouseEnter={() => {
            if (enableHoverOpen) setIsOpen(true);
          }}
          onMouseLeave={() => {
            if (enableHoverOpen) setIsOpen(false);
          }}
        >
          <li className="px-4 py-3 cursor-pointer hover:text-blue-600">Home</li>
          <li className="px-4 py-3 cursor-pointer hover:text-blue-600">About</li>
          <li className="px-4 py-3 cursor-pointer hover:text-blue-600">Services</li>
          <li className="px-4 py-3 cursor-pointer hover:text-blue-600">Contact</li>
        </ul>

      </div>
    </nav>
  );
}
```
